#include <iostream>
using namespace std;

int main() {
    int dayOfWeek;
    cout << "Provide value of any day of week, without weekend (1 = Monday)." << endl;
    cin >> dayOfWeek;
    switch (dayOfWeek)
    {
        case 1:
            cout << "Monday" << endl;
            break;
        case 2:
            cout << "Tuesday" << endl;
            break;
        case 3:
            cout << "Wednesday" << endl;
            break;
        case 4:
            cout << "Thursday" << endl;
            break;
        case 5:
            cout << "Friday" << endl;
            break;
        default:
            cout << "Incorrect value!" << endl;
            break;
    }
}
